import {getAllLearningBookmarks} from "../../lib/api.js"
import { useSSG } from "nextra/ssg";
import {HiExternalLink} from 'react-icons/hi'

export const getStaticProps = async ({params}) => {
    const data = await getAllLearningBookmarks()
    return {
        props: {
            ssg: {
                data
            }
        }
    }

}

export const Learning = () => {
    const {data} = useSSG()
    return (
        <div className="grid grid-cols-1 md:grid-cols-2 gap-4 mt-8">
            {
                data.map((item) => (
                <div className="flex flex-col border-2 rounded border-gray-400 dark:border-gray-600 p-2">
                    <h3 className="m-0">{item.name}</h3>
                    <span className="text-gray-500">{item.description}</span>
                    <a href={item.link} className="flex items-center" target="_blank">
                        <span>Link</span> 
                        <HiExternalLink/>
                    </a>
                </div>
                ))
            }
        </div>
    )

}

<Learning />
